# Plot y=x^2 
import numpy as np
import matplotlib.pyplot as plt
x = np.linspace (0,2,100)
fig, ax = plt.subplots()
y1 = np.sin(x)
y2 = np.cos(x)
plt.plot(x, y1, label='sin(x)', color='blue')
plt.plot(x, y2, label='cos(x)', color='red')
ax.set_xlabel("x")
ax.set_ylabel("y")
plt.legend()
plt.show()
